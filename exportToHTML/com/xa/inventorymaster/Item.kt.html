8    enum class StorageLocation {
9        KITCHEN_A, KITCHEN_B, KITCHEN_C, KITCHEN_D, KITCHEN_E, KITCHEN_F,
10       COFFEE_SHELF, GUEST_LOFT, MAIN_LOFT
11   }
12   
13   class Converters {
14       @TypeConverter
15       fun fromStorageLocation(value: StorageLocation?): String? {
16           return value?.name
17       }
18   
19       @TypeConverter
20       fun toStorageLocation(value: String?): StorageLocation? {
21           return value?.let { StorageLocation.valueOf(it) }
22       }
23   }
24
