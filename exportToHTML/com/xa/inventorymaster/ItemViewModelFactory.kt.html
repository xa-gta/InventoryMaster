9    class ItemViewModelFactory(
10       private val application: Application,

13       override fun <T : ViewModel> create(
14           key: String,
15           modelClass: Class<T>,
16           handle: SavedStateHandle
17       ): T {
18           if (modelClass.isAssignableFrom(ItemViewModel::class.java)) {
19               @Suppress("UNCHECKED_CAST")

23       }
