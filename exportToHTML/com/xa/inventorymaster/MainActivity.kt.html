35       private val pickPhotoLauncher = registerForActivityResult(ActivityResultContracts.StartActivityForResult()) { result ->
36           if (result.resultCode == Activity.RESULT_OK) {
37               result.data?.data?.let { uri ->
38                   val file = File(cacheDir, "item_photo_${System.currentTimeMillis()}.jpg")
39                   contentResolver.openInputStream(uri)?.use { input ->
40                       file.outputStream().use { output ->
41                           input.copyTo(output)
42                       }
43                   }
44                   selectedPhotoPath = file.absolutePath
45                   dialog?.findViewById<android.widget.ImageView>(R.id.photoPreview)?.setImageURI(uri)
46               }
47           }
48       }

96       private fun showAddItemDialog() {
97           dialog?.dismiss()
98   
99           val dialogBinding = DialogAddItemBinding.inflate(layoutInflater)
100          dialog = AlertDialog.Builder(this)
101              .setView(dialogBinding.root)
102              .setCancelable(true)
103              .create()
104  
105          // Set up the storage location spinner
106          val storageLocations = StorageLocation.values().map { it.name.replace("_", " ") }
107          val adapter = android.widget.ArrayAdapter(
108              this,
109              android.R.layout.simple_spinner_item,
110              storageLocations
111          )
112          adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item)
113          dialogBinding.storageLocationSpinner.adapter = adapter
114  
115          // Restore saved form data
116          dialogBinding.nameEditText.setText(viewModel.getFormName())
117          dialogBinding.priceEditText.setText(viewModel.getFormPrice())
118          dialogBinding.quantityBoughtEditText.setText(viewModel.getFormQuantityBought())
119          dialogBinding.quantityRemainingEditText.setText(viewModel.getFormQuantityRemaining())
120          val storagePosition = storageLocations.indexOf(viewModel.getFormStorageLocation()?.replace("_", " "))
121          dialogBinding.storageLocationSpinner.setSelection(if (storagePosition != -1) storagePosition else 0)
122          dialogBinding.safetyStockEditText.setText(viewModel.getFormSafetyStock())
123          val unitPosition = resources.getStringArray(R.array.units).indexOf(viewModel.getFormUnit())
124          dialogBinding.unitSpinner.setSelection(if (unitPosition != -1) unitPosition else 0)
125  
126          // Set up photo picker
127          dialogBinding.pickPhotoButton.setOnClickListener {
128              val intent = Intent(Intent.ACTION_PICK, MediaStore.Images.Media.EXTERNAL_CONTENT_URI)
129              pickPhotoLauncher.launch(intent)
130          }
131  
132          // Add TextWatchers and OnItemSelectedListener
133          dialogBinding.nameEditText.addTextChangedListener(object : android.text.TextWatcher {
134              override fun beforeTextChanged(s: CharSequence?, start: Int, count: Int, after: Int) {}
135              override fun onTextChanged(s: CharSequence?, start: Int, before: Int, count: Int) {}
136              override fun afterTextChanged(s: android.text.Editable?) {
137                  viewModel.updateFormField(ItemViewModel.KEY_NAME, s.toString().trim())
138              }
139          })
140  
141          dialogBinding.priceEditText.addTextChangedListener(object : android.text.TextWatcher {
142              override fun beforeTextChanged(s: CharSequence?, start: Int, count: Int, after: Int) {}
143              override fun onTextChanged(s: CharSequence?, start: Int, before: Int, count: Int) {}
144              override fun afterTextChanged(s: android.text.Editable?) {
145                  viewModel.updateFormField(ItemViewModel.KEY_PRICE, s.toString())
146              }
147          })
148  
149          dialogBinding.quantityBoughtEditText.addTextChangedListener(object : android.text.TextWatcher {
150              override fun beforeTextChanged(s: CharSequence?, start: Int, count: Int, after: Int) {}
151              override fun onTextChanged(s: CharSequence?, start: Int, before: Int, count: Int) {}
152              override fun afterTextChanged(s: android.text.Editable?) {
153                  viewModel.updateFormField(ItemViewModel.KEY_QUANTITY_BOUGHT, s.toString())
154              }
155          })
156  
157          dialogBinding.quantityRemainingEditText.addTextChangedListener(object : android.text.TextWatcher {
158              override fun beforeTextChanged(s: CharSequence?, start: Int, count: Int, after: Int) {}
159              override fun onTextChanged(s: CharSequence?, start: Int, before: Int, count: Int) {}
160              override fun afterTextChanged(s: android.text.Editable?) {
161                  viewModel.updateFormField(ItemViewModel.KEY_QUANTITY_REMAINING, s.toString())
162              }
163          })
164  
165          dialogBinding.storageLocationSpinner.onItemSelectedListener = object : android.widget.AdapterView.OnItemSelectedListener {
166              override fun onItemSelected(parent: android.widget.AdapterView<*>?, view: android.view.View?, position: Int, id: Long) {
167                  val selectedLocation = parent?.getItemAtPosition(position).toString().replace(" ", "_")
168                  viewModel.updateFormField(ItemViewModel.KEY_STORAGE_LOCATION, selectedLocation)
169              }
170              override fun onNothingSelected(parent: android.widget.AdapterView<*>?) {}
171          }
172  
173          dialogBinding.safetyStockEditText.addTextChangedListener(object : android.text.TextWatcher {
174              override fun beforeTextChanged(s: CharSequence?, start: Int, count: Int, after: Int) {}
175              override fun onTextChanged(s: CharSequence?, start: Int, before: Int, count: Int) {}
176              override fun afterTextChanged(s: android.text.Editable?) {
177                  viewModel.updateFormField(ItemViewModel.KEY_SAFETY_STOCK, s.toString())
178              }
179          })
180  
181          dialogBinding.unitSpinner.onItemSelectedListener = object : android.widget.AdapterView.OnItemSelectedListener {
182              override fun onItemSelected(parent: android.widget.AdapterView<*>?, view: android.view.View?, position: Int, id: Long) {
183                  val selectedUnit = parent?.getItemAtPosition(position).toString()
184                  viewModel.updateFormField(ItemViewModel.KEY_UNIT, selectedUnit)
185              }
186              override fun onNothingSelected(parent: android.widget.AdapterView<*>?) {}
187          }
188  
189          // Adjust dialog window to handle soft input
190          dialog?.window?.setSoftInputMode(android.view.WindowManager.LayoutParams.SOFT_INPUT_ADJUST_RESIZE)
191          dialog?.window?.setLayout(
192              android.view.WindowManager.LayoutParams.MATCH_PARENT,
193              android.view.WindowManager.LayoutParams.WRAP_CONTENT
194          )
195          val displayMetrics = resources.displayMetrics
196          val maxHeight = (displayMetrics.heightPixels * 0.7).toInt()
197          dialog?.window?.setAttributes(
198              dialog?.window?.attributes?.apply {
199                  height = maxHeight
200              }
201          )
202  
203          dialogBinding.nameEditText.setOnFocusChangeListener { _, hasFocus ->
204              if (hasFocus) {
205                  dialogBinding.nestedScrollView.post {
206                      dialogBinding.nestedScrollView.smoothScrollTo(0, dialogBinding.nameEditText.top)
207                  }
208              }
209          }
210  
211          dialogBinding.addButton.setOnClickListener {
212              Log.d("MainActivity", "Add button clicked")
213              val name = dialogBinding.nameEditText.text.toString().trim()
214              val priceStr = dialogBinding.priceEditText.text.toString()
215              val quantityBoughtStr = dialogBinding.quantityBoughtEditText.text.toString()
216              val quantityRemainingStr = dialogBinding.quantityRemainingEditText.text.toString()
217              val storageLocationStr = dialogBinding.storageLocationSpinner.selectedItem.toString().replace(" ", "_")
218              val safetyStockStr = dialogBinding.safetyStockEditText.text.toString()
219              val unit = dialogBinding.unitSpinner.selectedItem.toString()
220  
221              if (name.isEmpty()) {
222                  dialogBinding.nameEditText.error = "Required"
223                  return@setOnClickListener
224              }
225  
226              val price = priceStr.toDoubleOrNull() ?: 0.0
227              val quantityBought = quantityBoughtStr.toIntOrNull() ?: 0
228              val quantityRemaining = quantityRemainingStr.toIntOrNull() ?: 0
229              val safetyStock = safetyStockStr.toIntOrNull() ?: 0
230              val storageLocation = StorageLocation.valueOf(storageLocationStr)
231  
232              kotlinx.coroutines.CoroutineScope(kotlinx.coroutines.Dispatchers.Main).launch {
233                  try {
234                      Log.d("MainActivity", "Checking for existing item with name: $name")
235                      val existingItem = withContext(kotlinx.coroutines.Dispatchers.IO) {
236                          viewModel.getAllItems().find { it.name.equals(name, ignoreCase = true) }
237                      }
238                      if (existingItem != null) {
239                          AlertDialog.Builder(this@MainActivity)
240                              .setTitle("Duplicate Item")
241                              .setMessage("An item named '$name' already exists. Do you want to edit it?")
242                              .setPositiveButton("Edit") { _, _ ->
243                                  dialogBinding.nameEditText.setText(existingItem.name)
244                                  dialogBinding.priceEditText.setText(existingItem.price.toString())
245                                  dialogBinding.quantityBoughtEditText.setText(existingItem.quantityBought.toString())
246                                  dialogBinding.quantityRemainingEditText.setText(existingItem.quantityRemaining.toString())
247                                  dialogBinding.storageLocationSpinner.setSelection(
248                                      storageLocations.indexOf(existingItem.storageLocation.name.replace("_", " "))
249                                  )
250                                  dialogBinding.safetyStockEditText.setText(existingItem.safetyStock.toString())
251                                  dialogBinding.unitSpinner.setSelection(
252                                      resources.getStringArray(R.array.units).indexOf(existingItem.weight)
253                                  )
254                                  selectedPhotoPath = existingItem.photoPath
255                                  if (selectedPhotoPath != null) {
256                                      dialogBinding.photoPreview.setImageBitmap(android.graphics.BitmapFactory.decodeFile(selectedPhotoPath))
257                                  }
258                              }
259                              .setNegativeButton("Cancel") { _, _ ->
260                                  dialog?.dismiss()
261                                  isDialogOpen = false
262                              }
263                              .show()
264                      } else {
265                          val newItem = Item(
266                              name = name,
267                              price = price,
268                              quantityBought = quantityBought,
269                              quantityRemaining = quantityRemaining,
270                              storageLocation = storageLocation,
271                              photoPath = selectedPhotoPath,
272                              weight = unit,
273                              safetyStock = safetyStock
274                          )
275                          withContext(kotlinx.coroutines.Dispatchers.IO) {
276                              viewModel.insert(newItem)
277                          }
278                          viewModel.clearFormData()
279                          dialogBinding.nameEditText.setText("")
280                          dialogBinding.priceEditText.setText("")
281                          dialogBinding.quantityBoughtEditText.setText("")
282                          dialogBinding.quantityRemainingEditText.setText("")
283                          dialogBinding.storageLocationSpinner.setSelection(0)
284                          dialogBinding.safetyStockEditText.setText("")
285                          dialogBinding.unitSpinner.setSelection(0)
286                          dialogBinding.photoPreview.setImageResource(R.drawable.ic_item_default)
287                          selectedPhotoPath = null
288                          dialog?.dismiss()
289                          isDialogOpen = false
290                      }
291                  } catch (e: Exception) {
292                      Log.e("MainActivity", "Error inserting new item: ${e.message}", e)
293                  }
294              }
295          }
296  
297          dialogBinding.cancelButton.setOnClickListener {
298              viewModel.clearFormData()
299              dialogBinding.nameEditText.setText("")
300              dialogBinding.priceEditText.setText("")
301              dialogBinding.quantityBoughtEditText.setText("")
302              dialogBinding.quantityRemainingEditText.setText("")
303              dialogBinding.storageLocationSpinner.setSelection(0)
304              dialogBinding.safetyStockEditText.setText("")
305              dialogBinding.unitSpinner.setSelection(0)
306              dialogBinding.photoPreview.setImageResource(R.drawable.ic_item_default)
307              selectedPhotoPath = null
308              dialog?.dismiss()
309              isDialogOpen = false
310          }
311  
312          dialog?.setOnDismissListener {
313              viewModel.saveFormData(
314                  name = dialogBinding.nameEditText.text.toString().trim(),
315                  price = dialogBinding.priceEditText.text.toString(),
316                  quantityBought = dialogBinding.quantityBoughtEditText.text.toString(),
317                  quantityRemaining = dialogBinding.quantityRemainingEditText.text.toString(),
318                  storageLocation = dialogBinding.storageLocationSpinner.selectedItem.toString().replace(" ", "_"),
319                  safetyStock = dialogBinding.safetyStockEditText.text.toString(),
320                  unit = dialogBinding.unitSpinner.selectedItem.toString()
321              )
322          }
323  
324          dialog?.show()
325          dialog?.window?.setWindowAnimations(R.style.DialogAnimation)
326      }
